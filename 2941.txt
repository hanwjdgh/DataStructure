#include <stdio.h>
#include <string.h>


int main(void){
	char str[101];
	int a = 0, len, sum = 0;

	scanf("%s", str);
	len = strlen(str);
	while (1){
		if (a == len)
			break;
		if (str[a] == 'c'){
			if (str[a + 1] == '='){
				++sum;
				a += 2;
				continue;
			}
			else if (str[a + 1] == '-'){
				++sum;
				a += 2;
				continue;
			}
		}
		else if (str[a] == 'd'){
			if (str[a + 1] == '-'){
				++sum;
				a += 2;
				continue;
			}
			else if (str[a + 1] == 'z'&&str[a + 2] == '='){
				++sum;
				a += 3;
				continue;
			}
		}
		else if (str[a] == 'l'&&str[a + 1] == 'j'){
			++sum;
			a += 2;
			continue;
		}
		else if (str[a] == 'n'&&str[a + 1] == 'j'){
			++sum;
			a += 2;
			continue;
		}
		else if (str[a] == 's'&&str[a + 1] == '='){
			++sum;
			a += 2;
			continue;
		}
		else if (str[a] == 'z'&&str[a + 1] == '='){
			++sum;
			a += 2;
			continue;
		}
		sum++;
		++a;


	}
	printf("%d", sum);
	return 0;
}
