#include <stdio.h>

void Kaprekar(int);
int arr[10000];
int main(void){
	int a;
	for (a = 0; a < 10000; a++){
		Kaprekar(a+1);
	}
	for (a = 0; a < 10000; a++){
		if (arr[a] != 1)
			printf("%d\n", a+1);
	}
	return 0;
}
void Kaprekar(int n){
	int sum = n;
	if (n >= 10000){
		sum += n / 10000;
		n %= 10000;
	}
	if (n >= 1000){
		sum += n / 1000;
		n %= 1000;
	}
	if (n >= 100){
		sum += n / 100;
		n %= 100;
	}
	if (n >= 10){
		sum += n / 10;
		n %= 10;
	}
	sum += n;
	arr[sum - 1] = 1;
}
