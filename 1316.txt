#include <stdio.h>
#include <string.h>

int main(void){
	int tcase,cnt = 0,arr[26];
	int a,b,len;
	char str[101];
	scanf("%d", &tcase);
	for (a = 0; a < tcase; a++){
		for (b = 0; b < 26; b++)
			arr[b] = 0;
		scanf("%s", str);
		len = strlen(str);
		for (b = 0; b < len; b++){
			if (arr[str[b] - 'a'] != 0 && str[b] != str[b - 1]){
				cnt++;
				break;
			}
			else
				arr[str[b] - 'a']++;
		}
	}
	printf("%d", tcase - cnt);
	return 0;
}
