#include <stdio.h>

int main(void){
	int tcase, val, nav;
	int a;
	int H, W, N;
	scanf("%d", &tcase);
	for (a = 0; a < tcase; a++){
		scanf("%d %d %d", &H, &W, &N);
		val = N / H;
		nav = N%H;
		if (nav != 0)
			printf("%d\n", 101 + val + ((nav - 1) * 100));
		else{
			printf("%d\n", 101 + (val-1) + ((H - 1) * 100));
		}
	}
	return 0;
}
