#include <stdio.h>
#include <stdlib.h>

int org[100000], tmp[100000], arr[100000];
char pm[200000];
int t = -1, bt = -1, count = 0;

void push(int);
int pop();
int main(void){
	int numofC,check=0;
	int a,val,cnt=0,b=0;
	scanf("%d", &numofC);

	for (a = 0; a<numofC; ++a){
		org[a] = a + 1;
	}
	for (a = 0; a < numofC; ++a){
		scanf("%d", &val);
		tmp[a] = val;
	}

	while (check<numofC*2){
		if (t>-1&&arr[t] == tmp[cnt]){
			cnt++;
			pop();
			++check;
			continue;
		}
		if (b < numofC)
			push(org[b++]);
		++check;
	}

	if (count == (numofC * 2)){
		for (a = 0; a < numofC * 2; a++)
			printf("%c\n", pm[a]);
	}
	else
		printf("NO\n");
	return 0;
}

void push(int n){
	arr[++t] = n;
	pm[count++] = '+';
}
int pop(){
	pm[count++] = '-';
	return arr[t--];
}
