#include<stdio.h>
int main(void) {
	int i, j, m, n, a[246913] = { 0, 1 };
	int cnt = 0;
	while (1) {
		scanf("%d", &m);
		cnt = 0;
		if (m == 0)
			break;
		for (i = 2; i <= m; i++)
			for (j = 2; i*j <= 2 * m; j++)
				a[i*j] = 1;

		for (i = m+1; i <= 2 * m; i++)
			if (!a[i]) 
				++cnt;
			
		printf("%d\n", cnt);
	}
	return 0;
}
