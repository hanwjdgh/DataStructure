#include <stdio.h>
#include <math.h>

int arr[300];
int sum[300];
int Max(int, int);
int main(){
	int num;
	int a, cn = 0;

	scanf("%d", &num);
	for (a = 0; a < num; a++){
		scanf("%d", &arr[cn++]);
	}
	sum[0] = 0;
	for (a = 0; a < num; a++){
		if (a == 0)
			sum[a+1] = arr[a];
		else if (a == 1)
			sum[a+1] = Max(arr[a - 1] + arr[a], arr[a]);
		else
			sum[a+1] = Max(sum[a-2]+arr[a - 1] + arr[a], sum[a-1] + arr[a]);
	}
	printf("%d", sum[num]);
	return 0;
}

int Max(int a, int b){
	if (a > b)
		return a;
	else
		return b;
}
