#include <stdio.h>

int main(void){
	int N, x = 1, y = 1;
	int a = 0, b = 0;
	int c = 1, dir = 0;

	scanf("%d", &N);
	if (N==1)
		printf("%d/%d\n", a + 1, b + 1);
	else{
		while (1){
			if (a == 0 && dir == 0){
				b += y;
				dir = 1;
			}
			else if (dir == 1 && b != 0){
				a += x;
				b -= y;
			}
			else if (dir !=3  && b == 0){
				a += x;
				dir = 3;
			}
			else if (dir == 3 && a != 0){
				a -= x;
				b += y;
				if (a == 0)
					dir = 0;
			}
			if (c == N-1)
				break;
			++c;
		}
		printf("%d/%d\n", a + 1, b + 1);
	}
	return 0;
}
