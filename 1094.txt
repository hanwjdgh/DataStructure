#include <stdio.h>

int main(void){
	int num, cnt = 0,stk=64;
	int sum=0,tstk;
	scanf("%d", &num);

	while (1){
		if (stk > num)
			stk /= 2;
		else if (stk < num){
			++cnt;
			sum += stk;
			tstk = stk;
			while (sum!=num){
				tstk /= 2;
				if (sum + tstk <= num){
					++cnt;
					sum += tstk;
				}
			}
			break;
		}
		else{
			++cnt;
			break;
		}
	}
	printf("%d", cnt);
	return 0;
}
