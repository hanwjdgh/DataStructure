#include <stdio.h>
#include <string.h>

int main(void){
	int num;
	int a,mul=10000000;
	int sum = 0;

	for (a = 0; a < 8; a++){
		scanf("%d", &num);
		sum += num*mul;
		mul /= 10;
	}
	if (sum == 12345678)
		printf("ascending\n");
	else if (sum == 87654321)
		printf("descending\n");
	else
		printf("mixed\n");
	return 0;
}
