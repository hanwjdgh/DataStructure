#include <stdio.h>

int main(void){
	int tcase, arr[15][14];
	int a, b;
	int k, n;
	scanf("%d", &tcase);

	for (a = 0; a < 15; a++){
		for (b = 0; b < 14; b++){
			if (a == 0)
				arr[a][b] = b + 1;
			else if (b == 0)
				arr[a][b] = 1;
			else{
				arr[a][b] = arr[a - 1][b] + arr[a][b - 1];
			}
		}
	}

	for (a = 0; a < tcase; a++){
		scanf("%d", &k);
		scanf("%d", &n);
		printf("%d\n", arr[k][n - 1]);
	}
	return 0;
}
