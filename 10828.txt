#include <stdio.h>
#include <string.h>
 
int main()
{
    int n;
    int re=0;
    int adding = 0;
    int top=0;
    scanf("%d",&n);
    char input[100]={};
    int stack[1000]={};
 
    while(n--)
    {
        scanf("%s",input);
 
        if(!strcmp("push",input)){
            top++;
            scanf("%d",&adding);
            stack[top] = adding;
        }
        else if(!strcmp("pop",input)){
            if(stack[top]==NULL) {printf("-1\n"); continue;}
            printf("%d\n",stack[top]);
            top--;
        }
        else if(!strcmp("size",input)){
            printf("%d\n",top);
        }
        else if(!strcmp("empty",input)){
            if(top == 0)
            {    printf("1\n"); continue;}
            printf("0\n");
        }
        else if(!strcmp("top",input)){
            if(stack[top] == 0){ printf("-1\n"); continue;}
            printf("%d\n",stack[top]);
        }
 
    }
}
