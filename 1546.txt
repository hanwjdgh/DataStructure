#include <stdio.h>

int main(){
	int N,a,MAx;
	int arr[1000];
	double tmp[1000], hap=0;
	scanf("%d", &N);
	for (a = 0; a < N; a++){
		scanf("%d", &arr[a]);
		if (a == 0)
			MAx = arr[a];
		else if (MAx < arr[a])
			MAx = arr[a];
	}
	for (a = 0; a < N; a++){
		tmp[a] = (double) arr[a] / MAx * 100;
		hap += tmp[a];
	}
	printf("%.2lf", hap / N);
	return 0;
}
