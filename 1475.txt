#include <stdio.h>

int main(void){
	int arr[10] = { 0, };
	int num,a,max;

	scanf("%d", &num);
	while (num > 0){
		if (num % 10 == 6 || num % 10 == 9){
			if (arr[6] == arr[9])
				arr[6]++;
			else{
				if (arr[6] < arr[9])
					arr[6]++;
				else
					arr[9]++;
			}
		}
		else
			arr[num % 10]++;
		num /= 10;
	}
	for (a = 0; a < 10; a++){
		if (a == 0)
			max = arr[a];
		else if (max < arr[a])
			max = arr[a];
	}
	printf("%d", max);
	return 0;
}
