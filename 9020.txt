#include<stdio.h>
#include<math.h>
#define MIN -1

int main(void) {
	int i, j, arr[10001] = { 0, 1 };
	int num,min=0,test,cnt=0;
	int t1, t2;

	for (i = 2; i <= 10000; i++)
		for (j = 2; i*j <= 10000; j++)
			arr[i*j] = 1;

	scanf("%d", &test);

	while (cnt<test){
		scanf("%d", &num);
		if (num%2==1)
			continue;

		else {
			min = MIN;
			for (i = 0; i <= num; i++) {
				if (arr[i] == 0 && arr[num - i] == 0) {
					if (min == MIN) {
						t1 = i;
						t2 = num - i;
						min = abs((num - i) - i);
					}
					else if (min > abs((num - i) - i)) {
						t1 = i;
						t2 = num - i;
						min = abs((num - i) - i);
					}
				}
			}
			printf("%d %d\n", t1, t2);
		}
		++cnt;
	}
	return 0;
}
