#include <stdio.h>
#include <string.h>

int main(void){
	int arr[26];
	char str[101];
	int a;
	scanf("%s", str);

	for (a = 0; a < 26; a++)
		arr[a] = -1;

	for (a = 0; a < strlen(str); a++){
		if (arr[str[a] - 'a'] == -1)
			arr[str[a] - 'a'] = a;
	}
	for (a = 0; a < 26; a++)
		printf("%d ", arr[a]);
	return 0;
}
