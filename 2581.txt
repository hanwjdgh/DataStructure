#include <stdio.h>

int main(void) {
	int M, N, min = 0;
	int a, b;
	int cnt, pri = 0, sum = 0;

	scanf("%d", &M);
	scanf("%d", &N);

	for (a = M; a <= N; a++) {
		if (a != 1) {
			cnt = 0;
			for (b = 1; b <= a; b++) {
				if (a % b == 0)
					cnt++;
			}
			if (cnt == 2) {
				if (sum == 0)
					min = a;
				sum += a;
			}
		}
	}
	if (sum == 0)
		printf("%d", -1);
	else {
		printf("%d\n", sum);
		printf("%d\n", min);
	}
	return 0;
}
