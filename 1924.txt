#include <stdio.h>

int main(void){
	int ca, tmp = 0;
	int mon, day;
	int a;

	scanf("%d %d", &mon, &day);
	if (mon == 1)
		ca = day % 7;
	else{
		for (a = 1; a < mon; a++){
			if (a == 2)
				tmp += 28;
			else if (a == 4 || a == 6 || a == 9 || a == 11)
				tmp += 30;
			else
				tmp += 31;
		}
		tmp += day;
		ca = tmp % 7;
	}
	switch (ca){
	case 1:
		printf("MON\n");
		break;
	case 2:
		printf("TUE\n");
		break;
	case 3:
		printf("WED\n");
		break;
	case 4:
		printf("TUR\n");
		break;
	case 5:
		printf("FRI\n");
		break;
	case 6:
		printf("SAT\n");
		break;
	case 0:
		printf("SUN\n\n");
		break;
	default:
		break;
	}
	return 0;
}
