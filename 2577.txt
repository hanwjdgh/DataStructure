#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main(void){
	int cnt[10] = { 0, };
	int a,tmp=1,num;
	for (a = 0; a < 3; a++){
		scanf("%d", &num);
		tmp *= num;
	}
	for (a = 0; tmp > 0; a++){
		cnt[tmp % 10]++;
		tmp /= 10;
	}
	for (a = 0; a < 10; a++)
		printf("%d\n", cnt[a]);
	return 0;
}
