#include <stdio.h>
#include <string.h>
#include <math.h>

int main(void){
	int arr[3] = { 0, };
	int num;
	int a,b,tmp,cnt;

	for (a = 0; a < 2; a++){
		scanf("%d", &num);
		tmp = num;
		cnt = 2;
		for (b = 0; b < 3; b++){
			arr[a] += (tmp % 10)*pow(10,cnt);
			tmp /= 10;
			--cnt;
		}
	}
	printf("%d", (arr[0]>arr[1]) ? arr[0] : arr[1]);
	return 0;
}
