#include <stdio.h>

int main(){
	int N, cnt = 0;
	int arr[2], tmp[2];
	scanf("%d", &N);

	arr[0] = N / 10;
	arr[1] = N % 10;
	
	while (1){
		tmp[0] = arr[1];
		tmp[1] = (arr[0] + arr[1])%10;
		++cnt;
		arr[0] = tmp[0];
		arr[1] = tmp[1];
		if (arr[0] * 10 + arr[1] == N)
			break;
	}
	printf("%d", cnt);
	return 0;
}
